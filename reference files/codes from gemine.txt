// Get the button element
const topBtn = document.getElementById("topBtn");

// When the user scrolls down 200px from the top, show the button
window.onscroll = function() {
    if (document.body.scrollTop > 200 || document.documentElement.scrollTop > 200) {
        topBtn.style.display = "block";
    } else {
        topBtn.style.display = "none";
    }
};

// When the user clicks on the button, scroll to the top of the document
topBtn.onclick = function() {
    // Smooth scrolling property (modern browser feature)
    window.scrollTo({
        top: 0,
        behavior: 'smooth' 
    });
    
    // Fallback for older browsers (instant scroll)
    document.body.scrollTop = 0; 
    document.documentElement.scrollTop = 0; 
};


// Get the toggle button and the dropdown menu elements
const toggleBtn = document.querySelector('.toggle-btn');
const dropMenu = document.querySelector('.dropmenu');

// When the toggle button is clicked, add/remove the 'open' class
toggleBtn.onclick = function () {
    dropMenu.classList.toggle('open');
};
